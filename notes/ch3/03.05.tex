\section*{Preliminary}
\begin{description}
\item[Goal] Derivatives of trig functions: $f(x)=\cos(3x)$,
  $g(x)=\sin(x^2)$, $h(x)=\sqrt{\tan(x)}$.
\item[Idea] Sketch graph of $\sin(x)$, draw derivative.
  \[
  \dfrac{d}{dx}(\sin(x)))=\cos(x)
  \]
\end{description}
