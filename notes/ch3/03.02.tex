\section*{Limit}
\begin{description}
\item[Definition] Look at limit definition of $\displaystyle
  \frac{d}{dx}\left(a^x\right)$ (factor out $a^x$)
\item[SAGE] Calculate $\displaystyle \lim_{h\to
    0}\frac{a^h-1}{h}=\ln(a)$.
\item[Rule] $\displaystyle \frac{d}{dx}\left(a^x\right) = \ln(a)\cdot a^x$.
\end{description}
