\section*{Preliminary}
\begin{description}
\item[Goal] Derivatives of inverse functions: $f(x)=\ln(x)$,
  $g(x)=\arcsin(x)$.
\item[Abstract] Differentiate
  \[
  f(f^{-1}(x))=x
  \]
  to get
  \[
  \frac{d}{dx}(f^{-1}(x))=\frac{1}{f'(f^{-1}(x))}.
  \]
\end{description}
